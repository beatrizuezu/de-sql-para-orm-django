# .filter() vs .get()

from core.produtos.models import Produto

# FILTER - lista todos os produtos com o id = 1
Produto.objects.filter(categoria_id=1)

orm = Produto.objects.filter(id=1)

#para visualizar de como ficou a query
print(orm.query)

type(orm.query)

# LOOKUPS
"""
exact, iexact, contains, icontains, in, gt, gte, lt, lte, startswith,
istartswith, endswith, iendswith, range, date, year, month, day, week,
week_day, time, hour, minute, second, isnull, search, regex, iregex
https://docs.djangoproject.com/en/1.11/ref/models/querysets/#field-lookups
"""

#lista os produtos com o id igual a 1
Produto.objects.filter(id__in=[1, 3 ,5])

orm =  Produto.objects.filter(id__in=[1, 3 ,5])
print(orm.query) # para visualizar a query

# JOIN

#lista os produtos com categoria de nome 'categoria_1'
Produto.objects.filter(categoria__nome='categoria_1')

orm = Produto.objects.filter(categoria__nome='categoria_1')

print(orm.query) # para visualizar a query
